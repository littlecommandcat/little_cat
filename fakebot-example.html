HttpCat程式範例

// Provider: command_cat

#load <message>
begin;
    msg = message.content;
    if {str.count{msg ' '} == 0};
        output{'請附帶一個http code'};
    endif;
    msg_list = str.split{msg ' '};
    code = list.get{msg_list 1};
    try;
        code = turntype{code int};
    except;
        output{'附帶的code必須為整數'};
    endtry;
    if {code < 100 or code > 599};
        output{'附帶的code範圍必須為100~599'};
    endif;
    output{'https://http.cat/' + code};
end;
